**Goal:** Ship the storage + lifecycle layer for user-uploaded illustrations — `Asset` table with two-phase upload, atomic quota reservation, reference-aware owner soft-delete, admin abuse-delete with durable retry, and a daily GC cron with three reconciliation sweeps. import type { PrismaClient } from "@prisma/client"

```ts
export const QUOTA_BYTES = 100 * 1024 * 1024 // 100 MB

/**
 * Idempotent — calling repeatedly is safe. Uses raw SQL because Prisma's
 * upsert performs a SELECT first; we want a single INSERT ... ON CONFLICT.
 */
export async function ensureQuotaRow(
  prisma: PrismaClient,
  userId: string,
): Promise<void> {
  await prisma.$executeRaw`
    INSERT INTO "UserAssetQuota" ("userId", "reservedBytes", "updatedAt")
    VALUES (${userId}, 0::bigint, now())
    ON CONFLICT ("userId") DO NOTHING
  `
}

/**
 * Returns true if the reservation succeeded, false if it would exceed the cap.
 * Postgres row-locks serialize concurrent UPDATEs against the same userId.
 */
export async function reserveBytes(
  prisma: PrismaClient,
  userId: string,
  bytes: number,
): Promise<boolean> {
  await ensureQuotaRow(prisma, userId)
  const affected = await prisma.$executeRaw`
    UPDATE "UserAssetQuota"
    SET "reservedBytes" = "reservedBytes" + ${bytes}::bigint,
        "updatedAt" = now()
    WHERE "userId" = ${userId}
      AND "reservedBytes" + ${bytes}::bigint <= ${QUOTA_BYTES}::bigint
  `
  return affected === 1
}

export async function releaseBytes(
  prisma: PrismaClient,
  userId: string,
  bytes: number,
): Promise<void> {
  await prisma.$executeRaw`
    UPDATE "UserAssetQuota"
    SET "reservedBytes" = GREATEST(0::bigint, "reservedBytes" - ${bytes}::bigint),
        "updatedAt" = now()
    WHERE "userId" = ${userId}
  `
}
```

```ts
import { put, del, head } from "@vercel/blob"

const RETRY_DELAYS_MS = [200, 1000, 5000] // 3 inline retries, total ~6.2s wall clock

export interface BlobPutResult {
  url: string
  pathname: string
}

export async function putBlob(
  key: string,
  body: Blob | ArrayBuffer | Buffer,
  contentType: string,
): Promise<BlobPutResult> {
  const result = await put(key, body, {
    access: "public",
    addRandomSuffix: false,
    contentType,
  })
  return { url: result.url, pathname: result.pathname }
}

/**
 * Attempts del() up to 4 times (1 immediate + 3 retries). Throws the last
 * error on total failure. Caller decides what state to record.
 */
export async function delBlobWithRetry(url: string): Promise<void> {
  let lastErr: unknown
  for (let attempt = 0; attempt <= RETRY_DELAYS_MS.length; attempt++) {
    try {
      await del(url)
      return
    } catch (e) {
      lastErr = e
      const delay = RETRY_DELAYS_MS[attempt]
      if (delay) await new Promise((r) => setTimeout(r, delay))
    }
  }
  throw lastErr
}

export async function blobExists(url: string): Promise<boolean> {
  try {
    await head(url)
    return true
  } catch {
    return false
  }
}
```

- [ ] **Step 1: Implement `lib/uploads/rate-limit.ts`**

```ts
const MINUTE_MS = 60_000
const DAY_MS = 24 * 60 * MINUTE_MS

const PER_MINUTE = 10
const PER_DAY = 50

interface Window {
  minute: number[]
  day: number[]
}

const buckets = new Map<string, Window>()

export interface RateLimitResult {
  ok: boolean
  retryAfterMs?: number
}

export function checkUploadRate(userId: string): RateLimitResult {
  const now = Date.now()
  const w = buckets.get(userId) ?? { minute: [], day: [] }
  w.minute = w.minute.filter((t) => now - t < MINUTE_MS)
  w.day = w.day.filter((t) => now - t < DAY_MS)

  if (w.minute.length >= PER_MINUTE) {
    buckets.set(userId, w)
    return { ok: false, retryAfterMs: MINUTE_MS - (now - w.minute[0]) }
  }
  if (w.day.length >= PER_DAY) {
    buckets.set(userId, w)
    return { ok: false, retryAfterMs: DAY_MS - (now - w.day[0]) }
  }

  w.minute.push(now)
  w.day.push(now)
  buckets.set(userId, w)
  return { ok: true }
}

// Test-only: reset state.
export function _resetRateLimits(): void {
  buckets.clear()
}
```

```ts
// app/api/me/uploads/route.ts
import { NextResponse, type NextRequest } from "next/server"
import { auth } from "@/lib/auth"
import { prisma } from "@/lib/prisma"
import { reserveBytes, releaseBytes, ensureQuotaRow } from "@/lib/uploads/quota"
import { putBlob } from "@/lib/uploads/blob"
import { checkUploadRate } from "@/lib/uploads/rate-limit"
import { randomUUID } from "node:crypto"

const MAX_BYTES = 4 * 1024 * 1024 // 4 MB
const ALLOWED_TYPES = new Set([
  "image/svg+xml",
  "image/png",
  "image/jpeg",
  "image/webp",
  "image/gif",
])

const EXT_FOR_TYPE: Record<string, string> = {
  "image/svg+xml": "svg",
  "image/png": "png",
  "image/jpeg": "jpg",
  "image/webp": "webp",
  "image/gif": "gif",
}

const PENDING_TTL_MS = 5 * 60 * 1000

export async function POST(req: NextRequest) {
  const session = await auth()
  if (!session?.user) return NextResponse.json({ error: "unauthenticated" }, { status: 401 })
  if (session.user.role !== "ADMIN" && session.user.role !== "CONTRIBUTOR") {
    return NextResponse.json({ error: "forbidden" }, { status: 403 })
  }
  const ownerId = session.user.id

  const lenHeader = req.headers.get("content-length")
  if (lenHeader && Number(lenHeader) > MAX_BYTES + 4096) {
    return NextResponse.json({ error: "oversize" }, { status: 413 })
  }

  const rate = checkUploadRate(ownerId)
  if (!rate.ok) {
    return NextResponse.json(
      { error: "rate-limited", retryAfterMs: rate.retryAfterMs },
      { status: 429 },
    )
  }

  const formData = await req.formData()
  const file = formData.get("file")
  if (!(file instanceof Blob)) {
    return NextResponse.json({ error: "no-file" }, { status: 400 })
  }
  if (!ALLOWED_TYPES.has(file.type)) {
    return NextResponse.json({ error: "bad-type", contentType: file.type }, { status: 415 })
  }
  if (file.size > MAX_BYTES) {
    return NextResponse.json({ error: "oversize", bytes: file.size }, { status: 413 })
  }

  const ext = EXT_FOR_TYPE[file.type]
  const assetId = randomUUID()
  const blobKey = `learn/${ownerId}/${assetId}.${ext}`

  // Phase 1: reserve quota + insert PENDING Asset row, single transaction.
  let phase1Ok = false
  try {
    await prisma.$transaction(async (tx) => {
      await ensureQuotaRow(tx as unknown as typeof prisma, ownerId)
      const reserved = await reserveBytes(tx as unknown as typeof prisma, ownerId, file.size)
      if (!reserved) {
        // Throw to roll back; caller catches and returns 413.
        throw new Error("QUOTA_EXCEEDED")
      }
      await tx.asset.create({
        data: {
          id: assetId,
          ownerId,
          blobUrl: null,
          blobKey,
          contentType: file.type,
          bytes: file.size,
          status: "PENDING",
          pendingExpiresAt: new Date(Date.now() + PENDING_TTL_MS),
        },
      })
    })
    phase1Ok = true
  } catch (e) {
    if (e instanceof Error && e.message === "QUOTA_EXCEEDED") {
      return NextResponse.json({ error: "quota-exceeded" }, { status: 413 })
    }
    return NextResponse.json({ error: "phase1-failed" }, { status: 500 })
  }

  // Phase 2: write blob outside any transaction.
  let blobUrl: string
  try {
    const buf = Buffer.from(await file.arrayBuffer())
    const result = await putBlob(blobKey, buf, file.type)
    blobUrl = result.url
  } catch {
    // Roll back: mark DELETED and release quota.
    await prisma.$transaction(async (tx) => {
      await tx.asset.update({
        where: { id: assetId },
        data: { status: "DELETED", deletedAt: new Date() },
      })
      await releaseBytes(tx as unknown as typeof prisma, ownerId, file.size)
    })
    return NextResponse.json({ error: "blob-write-failed" }, { status: 502 })
  }

  // Phase 3: promote PENDING -> ACTIVE.
  const promote = await prisma.asset.updateMany({
    where: { id: assetId, status: "PENDING" },
    data: { status: "ACTIVE", blobUrl, pendingExpiresAt: null },
  })
  if (promote.count !== 1) {
    // Janitor took the row mid-flight. Delete the now-orphan blob, leave the
    // existing row alone (it was cleaned up correctly).
    try {
      const { delBlobWithRetry } = await import("@/lib/uploads/blob")
      await delBlobWithRetry(blobUrl)
    } catch {
      /* tolerated; orphan sweep handles */
    }
    return NextResponse.json({ error: "promote-race" }, { status: 409 })
  }

  const asset = await prisma.asset.findUniqueOrThrow({ where: { id: assetId } })
  return NextResponse.json({
    id: asset.id,
    url: blobUrl,
    contentType: asset.contentType,
    bytes: asset.bytes,
    createdAt: asset.createdAt,
  })
}
```

```ts
// lib/uploads/article-unlink.ts
import { unified } from "unified"
import remarkParse from "remark-parse"
import remarkStringify from "remark-stringify"
import remarkDirective from "remark-directive"
import { visit, SKIP } from "unist-util-visit"
import type { Node } from "unist"

/**
 * Returns the article's content with every `:::figure` whose `src` equals
 * the target URL removed. Other directives are preserved verbatim.
 */
export function stripFigureByUrl(content: string, targetUrl: string): string {
  const tree = unified().use(remarkParse).use(remarkDirective).parse(content) as Node
  visit(tree, (node: any, index, parent) => {
    if (
      node.type === "containerDirective" &&
      node.name === "figure" &&
      node.attributes?.src === targetUrl &&
      parent &&
      typeof index === "number"
    ) {
      ;(parent as any).children.splice(index, 1)
      return [SKIP, index]
    }
  })
  return unified().use(remarkStringify).use(remarkDirective).stringify(tree as any) as string
}

export function hasAnyDirectives(content: string): boolean {
  const tree = unified().use(remarkParse).use(remarkDirective).parse(content) as Node
  let found = false
  visit(tree, (node: any) => {
    if (
      node.type === "containerDirective" || node.type === "leafDirective"
    ) {
      if (
        ["figure", "mermaid", "steps", "side-by-side", "callout"].includes(node.name)
      ) {
        found = true
      }
    }
  })
  return found
}
```

```ts
export async function DELETE(
  _req: NextRequest,
  { params }: { params: Promise<{ id: string }> },

  const { id } = await params
  const session = await auth()
  if (!session?.user || session.user.role !== "ADMIN") {
    return NextResponse.json({ error: "forbidden" }, { status: 403 })
  }

  const asset = await prisma.asset.findUnique({ where: { id } })
  if (!asset) return NextResponse.json({ error: "not-found" }, { status: 404 })

  if (!asset.blobUrl) {
    return NextResponse.json({ error: "no-blob-url" }, { status: 500 })
  }
  const blobUrl = asset.blobUrl

    where: { content: { contains: blobUrl } },

          orderBy: { versionNumber: "desc" },
          select: { versionNumber: true },

        await tx.articleVersion.create({

            articleId: article.id,
            versionNumber: v,
            title: updated.title,
            slug: updated.slug,
            content: stripped,
            summary: updated.summary,
            topicId: updated.topicId,
            tagSlugs: [],
            relatedProblemSlugs: [],
            publishedById: session.user.id,
          },
        })
        affectedArticles.push({ slug: updated.slug, hadDirectives: true, snapshotVersion: v })
      } else {
        affectedArticles.push({ slug: updated.slug, hadDirectives: true })
      }
    }
  })

  return NextResponse.json({
    assetId: id,
```

```ts
// app/api/cron/asset-gc/route.ts
import { NextResponse, type NextRequest } from "next/server"
import { prisma } from "@/lib/prisma"
import { delBlobWithRetry } from "@/lib/uploads/blob"
import { releaseBytes } from "@/lib/uploads/quota"

const SEVEN_DAYS_MS = 7 * 24 * 60 * 60 * 1000

export async function POST(req: NextRequest) {
  if (
    req.headers.get("authorization") !== `Bearer ${process.env.CRON_SECRET}`
  ) {
    return NextResponse.json({ error: "forbidden" }, { status: 403 })
  }

  const out = {
    tombstones: { deleted: 0, failed: 0, bytesReclaimed: 0 },
    deleting: { deleted: 0, failed: 0 },
    pending: { promoted: 0, removed: 0 },
    orphan: { skipped: true, deleted: 0 },
  }

  // Sweep 1a: expired DELETED tombstones (owner soft-deletes).
  const tombstoneCutoff = new Date(Date.now() - SEVEN_DAYS_MS)
  const tombstones = await prisma.asset.findMany({
    where: { status: "DELETED", deletedAt: { lt: tombstoneCutoff } },
    select: { id: true, blobUrl: true, ownerId: true, bytes: true },
  })
  for (const t of tombstones) {
    if (!t.blobUrl) {
      // PENDING that never got promoted; sweep 2 covers this case.
      continue
    }
    try {
      await delBlobWithRetry(t.blobUrl)
      await prisma.$transaction(async (tx) => {
        await tx.asset.delete({ where: { id: t.id } })
        await releaseBytes(tx as unknown as typeof prisma, t.ownerId, t.bytes)
      })
      out.tombstones.deleted++
      out.tombstones.bytesReclaimed += t.bytes
    } catch {
      out.tombstones.failed++
    }
  }

  return NextResponse.json({ sweep: "complete", result: out })
}
```
